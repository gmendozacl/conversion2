def create_periods(inicio, fin):
    periodos =  []
    ano_inicio = int(inicio[:4])
    mes_inicio = int(inicio[-2:])
    ano_fin = int(fin[:4])
    mes_fin = int(fin[-2:])
    mes=mes_inicio
    ano=ano_inicio
    while True:
        if mes+1==14:
            ano=ano+1
            mes=1
        if len(str(mes))==1:mess='0'+str(mes)
        else:mess=str(mes)
        periodos.append([str(ano),mess])
        if ano == ano_fin and mes==mes_fin:break
        mes=mes+1

    return periodos
